gsap.to('.text', {
  text: 'This is animated text paragraph',
  duration: 2,
  ease: 'power1.in',
});

Very simple text animation example


gsap.from('.text', {
  text: 'This text goes backwards',
  duration: 2,
  ease: 'power1',
});

changing direction to from, makes text disapear 


gsap.to('.text', {
  text: 'This is yoyo tween',
  duration: 3,
  ease: 'none',
  yoyo: true,
});
      


gsap.fromTo('.text',
  { 
    text: 'White',
    color: 'white'
  },
  {
    text: 'Green!',
    color: 'green',
    ease: 'none',
    yoyo: true,
    duration: 2,
  }
);

Here we use fromTo to tween between two text and their color
